Qu 7. Write a function that accepts parameter n and returns factorial of n?

    function factorial(n){
        let fact=1;
        for (let i=1;i<=n;i++){
          fact *= i;
        }
        return ("Factorial of "+n+" is "+fact);
    }
    console.log(factorial(5));
